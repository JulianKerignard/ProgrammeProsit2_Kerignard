class FibonacciProgram
{
    static void Main(string[] args)
    {
        Console.WriteLine("Suite de Fibonacci :");
        
        int n = 10; // nombre d'itérations a calculer 
        
        // Methode itérative
        Console.WriteLine("\nitérative :");
        for (int i = 0; i < n; i++)
        {
            Console.Write(FibonacciIterative(i) + " ");
        }
        
        // Affichage récursive
        Console.WriteLine("\n\nrécursive :");
        for (int i = 0; i < n; i++)
        {
            Console.Write(FibonacciRecursive(i) + " ");
        }
    }
    
    //itérative calculer Fibonacci
    static int FibonacciIterative(int n)
    {
        if (n == 0) return 0;
        if (n == 1) return 1;

        int a = 0;
        int b = 1;
        int fib = 0;

        for (int i = 2; i <= n; i++)
        {
            fib = a + b;
            a = b;
            b = fib;
        }

        return fib;
    }

    //récursive pour calculer Fibonacci
    static int FibonacciRecursive(int n)
    {
        if (n == 0) return 0;
        if (n == 1) return 1;

        return FibonacciRecursive(n - 1) + FibonacciRecursive(n - 2);
    }
}
